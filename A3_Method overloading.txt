class Calculator {

    // Overloaded power methods
    int power(int base, int exp) {
        return (int) Math.pow(base, exp);  // using Math.pow()
    }

    double power(double base, double exp) {
        return Math.pow(base, exp);       // using Math.pow()
    }

    // Overloaded absolute methods
    int absolute(int num) {
        return Math.abs(num);             // using Math.abs()
    }

    double absolute(double num) {
        return Math.abs(num);             // using Math.abs()
    }
}

public class OverloadingDemo {
    public static void main(String[] args) {

        Calculator calc = new Calculator();

        // Testing power methods
        System.out.println("Power (int): 2^5 = " + calc.power(2, 5));
        System.out.println("Power (double): 2.5^3 = " + calc.power(2.5, 3));

        // Testing absolute methods
        System.out.println("Absolute (int): |-15| = " + calc.absolute(-15));
        System.out.println("Absolute (double): |-12.78| = " + calc.absolute(-12.78));
    }
}
